ALTER PROCEDURE [dbo].[usp_CopySurvey]
( 
	@UserID INT,
	@CreatedUserID INT,
	@OldSurveyID INT,
	@NewSurveyName VARCHAR(200)
)
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
SET NOCOUNT ON;
DECLARE @NewSurveyID INT=0
Declare @QID int=0
Declare @OptId int=0
-- *** Checking if new survey name is alrady exists or not *** --
IF NOT EXISTS(SELECT Name FROM T_Manage_Surveys WHERE Name=@NewSurveyName AND UserID=@UserID) -- *** if not exists *** --
	BEGIN
		INSERT INTO T_Manage_Surveys ([ProfileID]
		  ,[UserID]
		  ,[Name]
		  ,[Description]
		  ,[STypeID]
		  ,[Thanks_Message]      
		  ,[IsDeleted]
		  ,[Created_Date]
		  ,[Created_User]
		  ,[Modified_Date]
		  ,[Modified_User])
		SELECT [ProfileID]
		  ,[UserID]
		  ,@NewSurveyName
		  ,[Description]
		  ,[STypeID]
		  ,[Thanks_Message]      
		  ,[IsDeleted]
		  ,GETDATE()
		  ,@CreatedUserID
		  ,GETDATE()
		  ,@CreatedUserID
		FROM T_Manage_Surveys WHERE Survey_ID= @OldSurveyID
		SET @NewSurveyID=SCOPE_IDENTITY()
		-- *** inserting questions based on old survey ID *** --
		INSERT INTO dbo.T_Survey_Questions  (Text,
			QTypeID, 
			SurveyID,
			Status,
			Created_Date,    
			Creadted_User,
			Modified_Date, 
			Modified_User)
		SELECT
			Text,  
			QTypeID,
			@NewSurveyID,
			Status,
			GETDATE(),
			@CreatedUserID,
			GETDATE(),
			@CreatedUserID
		FROM dbo.T_Survey_Questions TSQ  WHERE TSQ.SurveyID = @OldSurveyID
	CREATE TABLE #QuestionTable
	(
		Id int
	)
	INSERT INTO #QuestionTable SELECT Question_ID From T_Survey_Questions Where SurveyID=@OldSurveyID 
	CREATE TABLE #OptionsTable
	(
		Id int
	)
	INSERT INTO #OptionsTable SELECT Question_ID From T_Survey_Questions Where SurveyID=@NewSurveyID 
	/*******************************************************/
	While((Select Count(*) From #QuestionTable)>0)
	BEGIN
	Set @QID=(Select Top 1 Id From #QuestionTable)
	set @OptId=(Select Top 1 Id From #OptionsTable)
	INSERT INTO dbo.T_SurveyQuestion_Options  
	(  
		QuestionID,
		SurveyID, 
		Answer_Option,
		Order_No,
		IsDeleted,    
		Created_Date,
		Created_User, 
		Modified_Date,
		Modified_User)
	SELECT
		@OptId,  
		@NewSurveyID,
		Answer_Option,
		Order_No,
		IsDeleted,
		GETDATE(),
		@UserID,
		GETDATE(),
		@UserID

	FROM dbo.T_SurveyQuestion_Options TSQO  
	WHERE TSQO.SurveyID = @OldSurveyID AND QuestionID = @QID
	/*******************************************************/
	Delete From #QuestionTable Where Id=@QID
	Delete From #OptionsTable where Id=@OptId
	END
	Delete #QuestionTable
	Delete #OptionsTable
	END	
	SELECT @NewSurveyID
END

ALTER PROC [dbo].[usp_m_GetCallIndexButtons]
(
	@ProfileID INT,
	@UserModuleId INT = NULL
)
AS 
	IF(@UserModuleId IS NULL)
		BEGIN
			SELECT ROW_NUMBER() OVER(ORDER BY UserModuleID ASC) AS rowid ,* FROM T_UserCustom_Modules
				WHERE ProfileID=@ProfileID AND IsActive=1 AND ButtonType='PrivateCallAddOns'
		END
	ELSE
		BEGIN
			SELECT ROW_NUMBER() OVER(ORDER BY UserModuleID ASC) AS rowid ,* FROM T_UserCustom_Modules
				WHERE ProfileID=@ProfileID AND IsActive=1 AND ButtonType='PrivateCallAddOns' AND UserModuleID = @UserModuleId
		END
